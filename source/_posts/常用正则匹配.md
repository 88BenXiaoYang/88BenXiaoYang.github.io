* 匹配网址链接的正则表达式

```
+ (NSArray *)regexHttpFromText:(NSString *)text {
  //匹配网址链接
  //    NSString *regex_http = @"http(s)?://([a-zA-Z|\\d]+\\.)+[a-zA-Z|\\d]+(/[a-zA-Z|\\d|\\-|\\+|_./?%&=]*)?";
NSString *regex_http = @"((http|ftp|https):\\/\\/)?[\\w\\-_]+(\\.[\\w\\-_]+)+([\\w\\-\\.,@?^=%&amp;:/~\\+#]*[\\w\\-\\@?^=%&amp;/~\\+#])?";
return  [text componentsMatchedByRegex:regex_http];
}
```
